액세스 권한: 모든 사용자로 설정 
function doPost(e){
  const ss=SpreadsheetApp.getActiveSpreadsheet();
  const d=JSON.parse(e.postData.contents);
  if(d.action==='save'){
    let sh=ss.getSheetByName(d.mentor)||ss.insertSheet(d.mentor);
    sh.clearContents();
    sh.getRange(1,1).setValue(d.data);
    return ContentService.createTextOutput('ok');
  }
  if(d.action==='load'){
    let sh=ss.getSheetByName(d.mentor);
    if(!sh) return ContentService.createTextOutput('');
    return ContentService.createTextOutput(sh.getRange(1,1).getValue());
  }
}
